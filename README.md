function skill(name, costs, prereqs, opts){
  return {
    name,
    costs, // [M,R,T,N,A,W,Mg,Dr,Ba]
    frag: opts&&opts.frag ? opts.frag : 0,
    maxPurchases: opts&&opts.max ? opts.max : DEFAULT_MAX,
    bodyBonus: opts&&opts.body ? opts.body : 0,
    strBonus: opts&&opts.str ? opts.str : 0,
    occupational: opts&&opts.occ ? true : false,
    prereqPerPurchase: opts&&opts.prereqPerPurchase ? true : false,
    _weaponChoice: opts&&opts._weaponChoice ? opts._weaponChoice : null,
    prereqs: prereqs || [],
    desc: opts&&opts.desc ? opts.desc : 'Description coming soon.',
  };
}

// Helper for frag-only skills — fixed frag cost and CP cost, no per-occupation pricing
function fragSkill(name, frag, cp, prereqs, opts){
  return {
    name,
    costs: null, // no per-occupation pricing — cost is fixed
    cp: cp,
    frag: frag,
    maxPurchases: opts&&opts.max ? opts.max : DEFAULT_MAX,
    bodyBonus: opts&&opts.body ? opts.body : 0,
    strBonus: opts&&opts.str ? opts.str : 0,
    prereqPerPurchase: opts&&opts.prereqPerPurchase ? true : false,
    prereqs: prereqs || [],
    desc: opts&&opts.desc ? opts.desc : 'Description coming soon.',
    _fragOnly: true,
  };
}

// ─── Build unified searchable list ───────────────────
// Extract a flat list of skills from the SKILLS categories
function buildSkillsFlat() {
  const result = [];
  SKILLS.forEach(cat => {
    cat.skills.forEach(sk => {
      result.push({...sk, _type:'skill', cat: cat.cat});
    });
  });
  return result;
}

const ALL_ITEMS = [
  ...MONSTER_ABILITIES.map(a => ({...a, _type:'monster'})),
  ...SB_SPELLS.map(s => ({...s, _type:'spell', cat: s.sphere})),
  ...buildSkillsFlat(),
];

// ─── State ────────────────────────────────────────────
let rows = [];
let nextId = 1;
let activeDropdown = null;
let focusedIndex = -1;

// ─── Filter ───────────────────────────────────────────
function filterItems(query) {
  const q = (query || '').toLowerCase().trim();
  if (!q) return ALL_ITEMS;
  return ALL_ITEMS.filter(item => {
    const n = (item.name || '').toLowerCase();
    const c = (item.cat || '').toLowerCase();
    const d = (item.desc || '').toLowerCase();
    return n.includes(q) || c.includes(q) || d.includes(q);
  });
}

// ─── Row management ───────────────────────────────────
function addRow() {
  const id = nextId++;
  rows.push({ id, item: null, count: 1 });
  renderRows();
  setTimeout(() => {
    const el = document.getElementById('search-' + id);
    if (el) { el.focus(); openDropdown(id, ''); }
  }, 50);
}

function removeRow(id) {
  rows = rows.filter(r => r.id !== id);
  if (activeDropdown === id) activeDropdown = null;
  renderRows();
  updateSummary();
}

function setRowItem(id, item) {
  const row = rows.find(r => r.id === id);
  if (row) { row.item = item; updateSummary(); }
}

function setRowCount(id, val) {
  const row = rows.find(r => r.id === id);
  if (row) { row.count = Math.max(1, parseInt(val) || 1); updateSummary(); }
}

function renderRows() {
  const container = document.getElementById('rows-container');
  container.innerHTML = '';
  rows.forEach(row => {
    const div = document.createElement('div');
    div.className = 'ability-row';
    div.id = 'row-' + row.id;

    const searchWrap = document.createElement('div');
    searchWrap.className = 'search-wrap';

    const searchInput = document.createElement('input');
    searchInput.type = 'text';
    searchInput.className = 'search-input';
    searchInput.id = 'search-' + row.id;
    searchInput.autocomplete = 'off';
    searchInput.spellcheck = false;
    searchInput.placeholder = 'Search skills, spells, or monster abilities\u2026';
    if (row.item) searchInput.value = row.item.name;

    const dropdown = document.createElement('div');
    dropdown.className = 'dropdown-list';
    dropdown.id = 'dropdown-' + row.id;

    searchInput.addEventListener('input', () => {
      openDropdown(row.id, searchInput.value);
      if (!searchInput.value && row.item) { row.item = null; updateSummary(); }
    });
    searchInput.addEventListener('focus', () => openDropdown(row.id, searchInput.value));
    searchInput.addEventListener('keydown', e => {
      if (e.key === 'ArrowDown') { e.preventDefault(); moveFocus(1); }
      else if (e.key === 'ArrowUp') { e.preventDefault(); moveFocus(-1); }
      else if (e.key === 'Enter') { e.preventDefault(); selectFocused(row.id); }
      else if (e.key === 'Escape') { closeDropdown(); searchInput.blur(); }
    });

    searchWrap.appendChild(searchInput);
    searchWrap.appendChild(dropdown);

    const countInput = document.createElement('input');
    countInput.type = 'number';
    countInput.className = 'count-input';
    countInput.min = '1';
    countInput.value = row.count;
    countInput.addEventListener('change', () => setRowCount(row.id, countInput.value));
    countInput.addEventListener('input', () => setRowCount(row.id, countInput.value));

    const removeBtn = document.createElement('button');
    removeBtn.className = 'remove-btn';
    removeBtn.innerHTML = '\u2715';
    removeBtn.title = 'Remove this row';
    removeBtn.addEventListener('click', () => removeRow(row.id));

    div.appendChild(searchWrap);
    div.appendChild(countInput);
    div.appendChild(removeBtn);
    container.appendChild(div);
  });
}

// ─── Dropdown ─────────────────────────────────────────
function openDropdown(id, query) {
  if (activeDropdown && activeDropdown !== id) closeDropdownById(activeDropdown);
  activeDropdown = id;
  focusedIndex = -1;

  const dropdown = document.getElementById('dropdown-' + id);
  if (!dropdown) return;

  const filtered = filterItems(query);
  if (!filtered.length) {
    dropdown.innerHTML = '<div class="dropdown-none">No results found.</div>';
    dropdown.classList.add('open');
    dropdown._filtered = [];
    return;
  }

  const monsters = filtered.filter(i => i._type === 'monster');
  const spells   = filtered.filter(i => i._type === 'spell');
  const skills   = filtered.filter(i => i._type === 'skill');

  let html = '';
  let itemIdx = 0;

  function addGroup(label, items, badgeClass, badgeLabel) {
    if (!items.length) return;
    html += '<div class="dropdown-group-label">' + label + '</div>';
    items.forEach(item => {
      const level = item.level ? ' \u00b7 Circle ' + item.level : '';
      const cat   = item.cat   ? ' \u00b7 ' + item.cat : '';
      html += '<div class="dropdown-item" data-idx="' + itemIdx + '" '
        + 'onmousedown="selectItem(' + id + ',' + itemIdx + ')" '
        + 'onmouseover="previewItem(' + itemIdx + ')">'
        + '<span class="item-name">' + item.name + cat + level + '</span>'
        + '<span class="item-badge ' + badgeClass + '">' + badgeLabel + '</span>'
        + '</div>';
      itemIdx++;
    });
  }

  addGroup('Monster Abilities', monsters, 'badge-monster', 'Monster');
  addGroup('Spells',            spells,   'badge-spell',   'Spell');
  addGroup('Skills',            skills,   'badge-skill',   'Skill');

  dropdown.innerHTML = html;
  dropdown._filtered = filtered;
  dropdown.classList.add('open');
}

function closeDropdown() {
  if (activeDropdown) { closeDropdownById(activeDropdown); activeDropdown = null; }
  focusedIndex = -1;
}

function closeDropdownById(id) {
  const d = document.getElementById('dropdown-' + id);
  if (d) d.classList.remove('open');
}

function moveFocus(dir) {
  if (!activeDropdown) return;
  const dd = document.getElementById('dropdown-' + activeDropdown);
  if (!dd) return;
  const items = dd.querySelectorAll('.dropdown-item');
  if (!items.length) return;
  if (items[focusedIndex]) items[focusedIndex].classList.remove('focused');
  focusedIndex = Math.max(0, Math.min(items.length - 1, focusedIndex + dir));
  if (items[focusedIndex]) {
    items[focusedIndex].classList.add('focused');
    items[focusedIndex].scrollIntoView({ block: 'nearest' });
  }
  if (dd._filtered && dd._filtered[focusedIndex]) showDetail(dd._filtered[focusedIndex]);
}

function selectFocused(id) {
  if (focusedIndex < 0) return;
  const dd = document.getElementById('dropdown-' + id);
  if (!dd || !dd._filtered) return;
  const item = dd._filtered[focusedIndex];
  if (item) applySelection(id, item);
}

function selectItem(id, idx) {
  const dd = document.getElementById('dropdown-' + id);
  if (!dd || !dd._filtered) return;
  const item = dd._filtered[idx];
  if (item) applySelection(id, item);
}

function applySelection(id, item) {
  setRowItem(id, item);
  const input = document.getElementById('search-' + id);
  if (input) input.value = item.name;
  closeDropdown();
  showDetail(item);
  updateSummary();
}

function previewItem(idx) {
  if (!activeDropdown) return;
  const dd = document.getElementById('dropdown-' + activeDropdown);
  if (!dd || !dd._filtered) return;
  const item = dd._filtered[idx];
  if (item) showDetail(item);
}

// ─── Detail panel ─────────────────────────────────────
function showDetail(item) {
  const panel = document.getElementById('detail-panel');
  if (!item) {
    panel.innerHTML = '<div class="detail-empty">Hover a dropdown item to see details</div>';
    return;
  }
  const tc = item._type === 'spell' ? 'spell' : item._type === 'monster' ? 'monster' : 'skill';
  const tl = item._type === 'spell' ? 'Spell'  : item._type === 'monster' ? 'Monster Ability' : 'Skill';
  const catTag   = item.cat   ? '<span class="detail-tag">' + item.cat + '</span>' : '';
  const lvlTag   = item.level ? '<span class="detail-tag">Circle ' + item.level + '</span>' : '';
  const callHtml = item.call   ? '<div class="detail-incant">\uD83D\uDCE2 ' + item.call + '</div>' : '';
  const incHtml  = item.incant ? '<div class="detail-incant">\uD83D\uDCDC ' + item.incant + '</div>' : '';
  panel.innerHTML = '<div class="detail-name">' + item.name + '</div>'
    + '<div class="detail-meta-row"><span class="detail-tag ' + tc + '">' + tl + '</span>' + catTag + lvlTag + '</div>'
    + callHtml + incHtml
    + '<div class="detail-desc">' + (item.desc || '') + '</div>';
}

// ─── Summary ──────────────────────────────────────────
function updateSummary() {
  const typeVal = document.getElementById('inp-type').value.trim();
  const bodyVal = document.getElementById('inp-body').value.trim();
  const apVal   = document.getElementById('inp-ap').value.trim();

  const sumType = document.getElementById('sum-type');
  sumType.textContent = typeVal || '\u2014';
  sumType.style.color      = typeVal ? '' : 'var(--color-text-tertiary)';
  sumType.style.fontStyle  = typeVal ? 'normal' : 'italic';
  sumType.style.fontWeight = typeVal ? '500' : '400';
  sumType.style.fontSize   = typeVal ? '16px' : '13px';

  document.getElementById('sum-body').textContent = bodyVal || '\u2014';
  document.getElementById('sum-ap').textContent   = apVal   || '\u2014';

  const tagsEl = document.getElementById('summary-tags');
  const filledRows = rows.filter(r => r.item);
  if (!filledRows.length) {
    tagsEl.innerHTML = '<span class="summary-empty-note">No skills, spells, or abilities added yet.</span>';
    return;
  }
  tagsEl.innerHTML = filledRows.map(r => {
    const tc = r.item._type === 'spell' ? 'is-spell' : r.item._type === 'monster' ? 'is-monster' : 'is-skill';
    const cnt = r.count > 1 ? '<span class="tag-count">\u00d7' + r.count + '</span>' : '';
    return '<span class="sum-tag ' + tc + '">' + r.item.name + cnt + '</span>';
  }).join('');
}

// ─── Print ────────────────────────────────────────────
function printMonster() {
  const typeVal = document.getElementById('inp-type').value.trim() || 'Unnamed Monster';
  const bodyVal = document.getElementById('inp-body').value.trim() || '\u2014';
  const apVal   = document.getElementById('inp-ap').value.trim()   || '\u2014';
  const filledRows = rows.filter(r => r.item);

  const monRows  = filledRows.filter(r => r.item._type === 'monster');
  const spRows   = filledRows.filter(r => r.item._type === 'spell');
  const skRows   = filledRows.filter(r => r.item._type === 'skill');

  function section(label, items) {
    if (!items.length) return '';
    const rowsHtml = items.map(r =>
      '<div style="display:flex;justify-content:space-between;padding:3px 0;border-bottom:0.5px dotted #ddd;font-size:12px">'
      + '<span>' + r.item.name + (r.count > 1 ? ' \u00d7' + r.count : '') + '</span>'
      + '<span style="font-size:10px;color:#666">' + (r.item.cat || r.item.sphere || '') + '</span>'
      + '</div>'
    ).join('');
    return '<div style="font-size:9px;text-transform:uppercase;letter-spacing:1px;border-bottom:1px solid #aaa;margin:12px 0 8px;padding-bottom:3px;color:#444">' + label + '</div>' + rowsHtml;
  }

  document.getElementById('print-area').innerHTML =
    '<div style="font-family:serif;padding:24px;color:#111">'
    + '<div style="font-size:20px;font-weight:bold;border-bottom:2px solid #333;margin-bottom:12px;padding-bottom:8px">' + typeVal + '</div>'
    + '<div style="display:flex;gap:28px;margin-bottom:14px">'
    + '<div style="display:flex;flex-direction:column"><span style="font-size:9px;text-transform:uppercase;letter-spacing:1px;color:#666">Body</span><span style="font-size:15px;font-weight:bold">' + bodyVal + '</span></div>'
    + '<div style="display:flex;flex-direction:column"><span style="font-size:9px;text-transform:uppercase;letter-spacing:1px;color:#666">Armour Points</span><span style="font-size:15px;font-weight:bold">' + apVal + '</span></div>'
    + '<div style="display:flex;flex-direction:column"><span style="font-size:9px;text-transform:uppercase;letter-spacing:1px;color:#666">Total Abilities</span><span style="font-size:15px;font-weight:bold">' + filledRows.length + '</span></div>'
    + '</div>'
    + section('Monster Abilities', monRows)
    + section('Spells', spRows)
    + section('Skills', skRows)
    + '</div>';

  window.print();
}

// ─── Global click to close dropdown ───────────────────
document.addEventListener('mousedown', e => {
  if (!activeDropdown) return;
  const wrap = document.getElementById('row-' + activeDropdown);
  if (wrap && !wrap.contains(e.target)) {
    const row = rows.find(r => r.id === activeDropdown);
    const inp = document.getElementById('search-' + activeDropdown);
    if (row && inp) inp.value = row.item ? row.item.name : '';
    closeDropdown();
  }
});
